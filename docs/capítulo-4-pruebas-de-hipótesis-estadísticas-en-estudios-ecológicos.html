Prueba t de Student

grupo_1 <- hormigas$Taxa_S[hormigas$TRATAMIENTO == "ECB"]
grupo_2 <- hormigas$Taxa_S[hormigas$TRATAMIENTO == "ECD"]

resultado <- t.test(grupo_1, grupo_2)
resultado
##
